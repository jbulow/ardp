103 /* ====================================================================== */
104 /* COPYRIGHT (C) 1986 Gary S. Brown. You may use this program, or */
105 /* code or tables extracted from it, as desired without restriction. */
106 /* */
107 /* First, the polynomial itself and its table of feedback terms. The */
108 /* polynomial is */
109 /* X^32+X^26+X^23+X^22+X^16+X^12+X^11+X^10+X^8+X^7+X^5+X^4+X^2+X^1+X^0 */
110 /* */
111 /* Note that we take it "backwards" and put the highest-order term in */
112 /* the lowest-order bit. The X^32 term is "implied"; the LSB is the */
113 /* X^31 term, etc. The X^0 term (usually shown as "+1") results in */
114 /* the MSB being 1. */
115 /* */
116 /* Note that the usual hardware shift register implementation, which */
117 /* is what we're using (we're merely optimizing it by doing eight-bit */
118 /* chunks at a time) shifts bits into the lowest-order term. In our */
119 /* implementation, that means shifting towards the right. Why do we */
120 /* do it this way? Because the calculated CRC must be transmitted in */
121 /* order from highest-order term to lowest-order term. UARTs transmit */
122 /* characters in order from LSB to MSB. By storing the CRC this way, */
123 /* we hand it to the UART in the order low-byte to high-byte; the UART */
124 /* sends each low-bit to hight-bit; and the result is transmission bit */
125 /* by bit from highest- to lowest-order term without requiring any bit */
126 /* shuffling on our part. Reception works similarly. */
127 /* */
128 /* The feedback terms table consists of 256, 32-bit entries. Notes: */
129 /* */
130 /* The table can be generated at runtime if desired; code to do so */
131 /* is shown later. It might not be obvious, but the feedback */
132 /* terms simply represent the results of eight shift/xor opera- */
133 /* tions for all combinations of data and CRC register values. */
134 /* */
135 /* The values must be right-shifted by eight bits by the "updcrc" */
136 /* logic; the shift must be unsigned (bring in zeroes). On some */
137 /* hardware you could probably optimize the shift in assembler by */
138 /* using byte-swap instructions. */
139 /* polynomial $edb88320 */
140 /* */

201 static unsigned int map_hash_int( map *m, char *keystring ) {
202  unsigned long key = crc32( ( unsigned char * )( keystring ), strlen( keystring ) );
203 
204  /* Robert Jenkins' 32 bit Mix Function */
205  key += ( key << 12 );
206  key ^= ( key >> 22 );
207  key += ( key << 4 );
208  key ^= ( key >> 9 );
209  key += ( key << 10 );
210  key ^= ( key >> 2 );
211  key += ( key << 7 );
212  key ^= ( key >> 12 );
213 
214  /* Knuth's Multiplicative Method */
215  key = ( key >> 3L ) * 2654435761L;
216 
217  return key % m->table_size;
218 }
